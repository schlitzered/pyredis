    class String(BaseCommand):
    """Mixin for Redis String and binary value commands (e.g. GET, SET, INCR, APPEND)."""

    def __init__(self):
        super().__init__()

    def append(self, *args):
        if self._cluster:
            return self.execute(
                *[b"APPEND", *args],
                shard_key=args[0]
            )
        return self.execute(
            *[b"APPEND", *args]
        )
